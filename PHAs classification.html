import numpy as np
pha_counts = df['pha'].value_counts()
hazardous_count = pha_counts['Y']
non_hazardous_count = pha_counts['N']
total_count = hazardous_count + non_hazardous_count

# Generate random indices for null values based on class proportions
random_indices = np.random.choice([0, 1], size=df['pha'].isnull().sum(), p=[non_hazardous_count/total_count, hazardous_count/total_count])

# Assign random values to the null instances in 'pha' column
df.loc[df['pha'].isnull(), 'pha'] = ['N' if index == 0 else 'Y' for index in random_indices]

# Handle null values in other columns
for column in df.columns:
    if column != 'pha' and df[column].isnull().sum() > 0:
        if df[column].dtype == 'object':
            df[column].fillna(df[column].mode().iloc[0], inplace=True)
        else:
            df[column].fillna(df[column].mean(), inplace=True)
